## 
Notifications display temporary messages to the user. They can be simple messages or more complex ones with titles, descriptions, sounds, and progress steps.

<Tabs items={['Simple', 'Advanced', 'Manipulation', 'Steps']}>
  <Tab value="Simple">
    ```lua
    -- Simple notification with a 5 second duration
    Library:Notify("This is a notification", 5)
    ```
  </Tab>
  <Tab value="Advanced">
    ```lua
    -- Advanced notification with more options
    Library:Notify({
        Title = "Success",
        Description = "Operation completed successfully",
        Time = 5, -- Duration in seconds
        SoundId = 123456789 -- Optional sound ID
    })
    ```
  </Tab>
  <Tab value="Manipulation">
    ```lua
    -- Get the notification object to update it later
    local Notification = Library:Notify({
        Title = "Progress",
        Description = "Processing...",
        Time = 10
    })

    -- Update notification text
    task.wait(2)
    Notification:ChangeTitle("Almost done")
    Notification:ChangeDescription("Finalizing...")
    ```
  </Tab>
  <Tab value="Steps">
    ```lua
    -- For notifications with steps
    local StepNotification = Library:Notify({
        Title = "Progress",
        Description = "Downloading files...",
        Steps = 5, -- Total number of steps
        Time = 20 -- or use a number for time-based
    })

    -- Update steps
    task.wait(2)
    StepNotification:ChangeStep(2) -- Set to step 2 of 5
    task.wait(2)
    StepNotification:ChangeStep(3) -- Set to step 3 of 5
    ```
  </Tab>
</Tabs>
